SELECT categories.name AS "category_name", blog_posts.title, users.name AS "author_name" FROM post_categories JOIN blog_posts ON post_categories.post_id = blog_posts.id JOIN users ON blog_posts.author_id = users.id JOIN categories ON post_categories.category_id = categories.id WHERE categories.name = "Basic";
SELECT categories.name AS "category_name", blog_posts.title,  blog_posts.body FROM post_categories JOIN blog_posts ON post_categories.post_id = blog_posts.id JOIN users ON blog_posts.author_id = users.id JOIN categories ON post_categories.category_id = categories.id WHERE users.name = "Robert" OR users.name = "Frank";
SELECT users.name AS "author_name", categories.name AS "category_name" FROM post_categories JOIN blog_posts ON post_categories.post_id = blog_posts.id JOIN users ON blog_posts.author_id = users.id JOIN categories ON post_categories.category_id = categories.id WHERE post_categories.post_id IN(SELECT post_id FROM post_categories GROUP BY post_id HAVING COUNT(category_id) >= 2) ORDER BY category_name ASC;
